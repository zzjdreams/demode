function y=PSK(t)
clc;clear all;

t=-10:0.01:10;
y=0;
st=[1,0,0,1,0,1];
for n=1:length(st)
    y=y+st(n).*(t> (n-1) & t<n);
end

subplot(5,1,1);
plot(t,y);
title('输入信源s（t）');
ylabel('s(t)');
xlabel('t(s)');
axis([0,6,-1,1]);

ys=0;
sta=-1*st+1;
for n=1:length(sta)
    ys=ys+sta(n).*(t> (n-1) & t<n);
end

subplot(5,1,2);
plot(t,ys);
title('s（t）翻转');
ylabel('s(t)');
xlabel('t(s)');
axis([0,length(st),-2,2]);


yc1=cos(30*t);
subplot(5,1,3);
plot(t,yc1);
title('cos(30*t)');
axis([0,length(st),-2,2]);



yc2=-yc1;
subplot(5,1,4);
plot(t,yc2);
title('cos(20*t)');
axis([0,length(st),-2,2]);


ycom=0;
ycom=y.*yc1+ys.*yc2;
subplot(5,1,5);
plot(t,ycom);
title('输出信号');
axis([0,length(st),-2,2]);


box;
% figure('Name','2FSK');

figure(1);
set(gcf,'Name','~~~~2FSK波形展示')
% figure('NumberTitle', 'off', 'Name', '2FSK波形展示');
set(gcf,'position',[300 50 900 700]);
end
